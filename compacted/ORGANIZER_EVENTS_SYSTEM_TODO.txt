  return supabase

// Save pairings
async saveEventPairings(eventId, pairingsData) {
  return supabase
    .from('event_pairings')
    .upsert({
      event_id: eventId,
      groups: pairingsData,
      locked_at: null,
      updated_at: new Date().toISOString()
    });
}

// Lock pairings
async lockPairings(eventId) {
  return supabase
    .from('event_pairings')
    .update({
      locked_at: new Date().toISOString(),
      locked_by: AppState.currentUser.lineUserId
    })
    .eq('event_id', eventId);
}
